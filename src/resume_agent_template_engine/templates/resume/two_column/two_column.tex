\newenvironment{twocolumnresume}{
    \columnratio{0.35,0.65}
    \setlength{\columnsep}{0.02\textwidth}

}{

}

% Sidebar section
\newenvironment{sidebar}{
    \switchcolumn[0]
    \color{sidebarText}
    \raggedright
}{
}

% Main content section
\newenvironment{maincontent}{

    \color{mainText}
    \raggedright
}{
}

% Sidebar section title
\newcommand{\sidebarsection}[1]{
    \vspace{0.3cm}
    {\large\bfseries\MakeUppercase{#1}}
    \vspace{0.2cm}
    \hrule
    \vspace{0.3cm}
}

% Main section title
\newcommand{\mainsection}[1]{
    \vspace{0.4cm}
    {\Large\bfseries\color{primaryColor}\MakeUppercase{#1}}
    \vspace{0.1cm}
    \hrule height 2pt
    \vspace{0.3cm}
}
